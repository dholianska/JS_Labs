Код JavaScript з використанням Стрілкових Функцій (Перевірте консоль)

Стрілкові функції автоматично успадковують контекст `this` від батьківської області видимості, що усуває необхідність у `bind()`. Це забезпечує коректну роботу з `this.data` та `this.result`.

let server = {
    data: 0,
    convertToString: function(callback) {
        // Стрілкова функція успадковує 'this' (server) від convertToString
        callback(() => {
            return this.data + "";
        });
    }
};

let client = {
    server: server,
    result: "",
    calc: function (data) {
        this.server.data = data;
        this.server.convertToString(this.notification());
    },
    notification: function() {
        // Стрілкова функція успадковує 'this' (client) від notification
        return (callback) => {
            this.result = callback();
        };
    }
};

client.calc(123);
// ...
// Результат у консолі: "123", "string"

У консолі ви побачите очікуваний вивід: "123" та "string".
